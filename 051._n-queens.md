class Solution(object):
    def solveNQueens(self, n):
        """
        :type n: int
        :rtype: List[List[str]]
        """
        self.board = []
        self.res = []
        self.placequeen(0,n)
        return self.res

    def danger(self, board ,row, col):
        for (i, j) in board:
            if row == i: return True
            if col == j: return True
            if abs(row - i) == abs(col - j): return True

        return False


    def placequeen(self, row, n):
        if row >= n:
            tmpRes = [['.' for i in range(n)] for j in range(n)]
            for (row,col) in self.board:
                tmpRes[row][col] = 'Q'
            tmpRes = [''.join(i) for i in tmpRes]
            self.res.append(tmpRes)

        else:
            for col in range(0, n):
                if not self.danger(self.board, row, col):
                    self.board.append((row, col))
                    self.placequeen(row + 1, n)
                    self.board.remove((row,col))

class Solution(object):
    def solveNQueens(self, n):
        """
        :type n: int
        :rtype: List[List[str]]
        """
        self.result=[]
        self.nums=[-1]*n
        self.dfs(self.nums,0)
        return self.result
    
    def dfs(self, nums, index):
        if index==len(nums):
            temp=[ ['.']*len(nums) for i in nums ]
            for row in range(len(nums)):
                temp[row][nums[row]]='Q'
            temp=[''.join(i) for i in temp]
            self.result.append(temp)
        
        else:
            for col in range(len(nums)):
                nums[index]=col
                if self.valid(nums, index):
                    self.dfs(nums,index+1)
                    
    def valid(self, nums, index):
        for i in range(index):
            if nums[i]==nums[index] or abs(nums[index]-nums[i])==index-i:
                return False
        return True
