```c++
#include <unordered_set>
class TwoSum
{
    private:
        std::unordered_set<int> seen;
    public:
        void add(int a);
        bool find(int a);
};

void TwoSum::add(int a)
{
    seen.insert(a);
}

bool TwoSum::find(int x)
{
    for(auto y:seen)
    {
    if( seen.find(x-y)!=seen.end() ) return true;
    }
    return false
}
```

```c++
class TwoSum {
public:
    void add(int number) {
        ++m[number];
    }
    bool find(int value) {
        for (auto a : m) {
            int t = value - a.first;
            if ((t != a.first && m.count(t)) || (t == a.first && a.second > 1)) {
                return true;
            }
        }
        return false;
    }
private:
    unordered_map<int, int> m;
};
```

Approach #3 Unordered_multiset
```c++class TwoSum {
public:
    void add(int number) {
        s.insert(number);
    }
    bool find(int value) {
        for (auto a : s) {
            int cnt = a == value - a ? 1 : 0;
            if (s.count(value - a) > cnt) {
                return true;
            }
        }
        return false;
    }
private:
    unordered_multiset<int> s;
};
```
