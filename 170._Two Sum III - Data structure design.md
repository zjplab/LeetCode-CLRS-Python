```c++
#include <unordered_set>
class TwoSum
{
    private:
        std::unordered_set<int> seen;
    public:
        void add(int a);
        bool find(int a);
};

void TwoSum::add(int a)
{
    seen.insert(a);
}

bool TwoSum::find(int x)
{
    for(auto y:seen)
    {
    if( seen.find(x-y)!=seen.end() ) return true;
    }
    return false
}
```
