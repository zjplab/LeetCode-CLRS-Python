class Solution:
    def check(self,t1,t2):
        if t1==None or t2==None:
            if t1==t2:
                return True
            else:
                return False
        if t1.val==t2.val:
            return self.check(t1.left,t2.right) and self.check(t1.right,t2.left)
        else:
            return False
        
    def isSymmetric(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        if root is None:
            return True
       
        else:
            return self.check(root.left,root.right)
